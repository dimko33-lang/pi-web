cat >> /opt/my-agent/agent.py << 'EOF'
    def _is_shell_command(self, text: str) -> bool:
        triggers = ["run","apt ","systemctl ","ls ","cd ","cat ","echo ","rm ","mkdir ","chmod ","chown ","grep ","find ","kill ","nano ","curl ","wget ","git clone","python3 ","pip "]
        for t in triggers:
            if text.lower().startswith(t): return True
        return False
    def _run_command(self, cmd: str) -> str:
        import subprocess
        try:
            out = subprocess.run(cmd, shell=True, capture_output=True, text=True, timeout=60)
            return (out.stdout+out.stderr).strip() or "OK"
        except Exception as e:
            return f"Error: {e}"
EOF
